def reshape_country(country):
    data = child_fertility.query("country == @country")
    long_data = data.melt(
        id_vars=["year", "country"],
        value_vars=["mortality", "woman_fertility"],
        var_name="Metric",
        value_name="Value",
    )
    return long_data
